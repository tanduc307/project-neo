datafile007_ns6

library(data.table)

datafile007_ns6_csv <- fread("US/3/datafile007-ns6.csv")

datafile007_ns6_csv_full <- transpose(datafile007_ns6_csv)

sample_rate <- 30000

signal_rate <- 1/sample_rate

dim(datafile007_ns6_csv_full)[1]*signal_rate -> sec

duration_1 <- seq(0, sec, length.out = dim(datafile007_ns6_csv_full)[1])

datafile007_ns6_csv_full$time_1 <- duration_1

datafile007_ns6_csv_full$time_1 -> x

datafile007_ns6_csv_full$V1 -> y_1 

plot(x, y_1, type = "l",
     col = 7,
     xaxs = "i", yaxs = "i",
     las = 1,
     xlab = "Time (s)",
     ylab = "Amplitude (\U03BCV)",
     xlim = c(0, 600), ylim = c(-2000, 2000),
     main = "Raw data from US | datafile007.ns6 | Channel ID: 1")
